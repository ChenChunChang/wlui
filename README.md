## Install
```shell
npm install weilan-ui -S  

# or

yarn add weilan-ui
```

## Quick Start

> 全部引入
``` js
import Vue from 'vue'
import WeilanUi from 'weilan-ui'
import 'weilan-ui/lib/weilan-ui.css';

Vue.use(WeilanUi)
```

> 按需引入
```js
import {
  Gantt, 
  Explorer, 
  BimViewer, 
  TransferTree, 
  FormDesigner
  FormParser
  Container, 
  ContextMenu, 
  FadeIn, 
  Scroll, 
  Address, 
  Add, 
  Input, 
  Select, 
  SelectTree, 
  Tree, 
  Table, 
  TableDynamic, 
} from 'weilan-ui'
import 'weilan-ui/lib/weilan-ui.css';

Vue.use(Gantt)
Vue.use(Explorer)
...
```
